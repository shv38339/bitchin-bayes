model{
  y.t[1] ~ dnorm(0, tau.stat)
  for(t in 2:ngames){
    y.t[t] ~ dnorm(yhat.t[t], tau)
    yhat.t[t] <- rho*y.t[t - 1]
  }
  y.p[1] ~ dnorm(yhat.p[1], tau)
  yhat.p[1] <- rho*y.t[ngames]
  for(p in 2:P){
    y.p[p] ~ dnorm(yhat.p[p], tau)
    yhat.p[p] <- rho*y.p[p-1]
  }
  tau.stat <- (1 - pow(rho, 2))/pow(sigma, 2)
  tau <- pow(sigma, -2)
  sigma ~ dunif(0, 15)
  rho ~ dunif(-1, 1)
}
